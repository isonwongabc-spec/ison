# OpenClaw 全新安装 - 终端执行代码
# 在 Windows PowerShell 或 CMD 中逐行执行

# ========== 第 1 步：安装 OpenClaw ==========
npm install -g openclaw-cn

# ========== 第 2 步：初始化配置 ==========
openclaw-cn gateway init

# ========== 第 3 步：配置 Telegram ==========
# 先去 @BotFather 创建机器人，获取 Token
openclaw-cn configure --section telegram
# 按提示输入：bot_token=你的Token

# ========== 第 4 步：配置 AI 模型（可选）==========
# 配置 OpenAI
openclaw-cn configure --section openai
# 按提示输入：api_key=你的Key

# 或配置其他模型（如 Claude、Kimi）
openclaw-cn configure --section anthropic
openclaw-cn configure --section moonshot

# ========== 第 5 步：配置搜索（可选）==========
# Brave 搜索
openclaw-cn configure --section brave
# 按提示输入：api_key=你的Key

# ========== 第 6 步：启动服务 ==========
openclaw-cn gateway start

# ========== 常用命令 ==========
# 查看状态
openclaw-cn status

# 停止服务
openclaw-cn gateway stop

# 重启服务
openclaw-cn gateway restart

# 查看帮助
openclaw-cn --help

# ========== 安装技能示例 ==========
# 搜索技能
openclaw-cn skills install brave-search

# 股票分析
openclaw-cn skills install stock-analyzer

# 加密货币
openclaw-cn skills install crypto-analyzer

# 查看已安装技能
openclaw-cn skills list
